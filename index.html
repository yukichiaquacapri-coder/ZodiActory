import swisheph from 'swisseph'; export default function handler(req, res) { // 画面から送られてきた日付（YYYY-MM-DD）と時間（HH:mm）を取得 const { date, time } = req.query; if (!date || !time) { return res.status(400).json({ error: "Date and time are required" }); } const [year, month, day] = date.split('-').map(Number); const [hour, minute] = time.split(':').map(Number); // ユリウス日を計算（スイスエフェメリス用） const julianDay = swisheph.swe_julday(year, month, day, hour + minute / 60, swisheph.SE_GREG_CAL); // 太陽(SE_SUN)の位置を計算 swisheph.swe_calc_ut(julianDay, swisheph.SE_SUN, swisheph.SEFLG_SPEED, (res_sun) => { const sunLong = res_sun.longitude; // 12星座のリスト const signs = ["牡羊座", "牡牛座", "双子座", "蟹座", "獅子座", "乙女座", "天秤座", "蠍座", "射手座", "山羊座", "水瓶座", "魚座"]; // 30度ずつで割って、どの星座か判定 const sunSign = signs[Math.floor(sunLong / 30)]; // 結果を返す res.status(200).json({ sun: `${sunSign} (${sunLong.toFixed(2)}°)`, moon: "計算中...", // 次に月やアセンダントも追加しましょう！ asc: "計算中..." }); }); }
