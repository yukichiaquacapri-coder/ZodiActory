async function calculate() { const resDiv = document.getElementById('result'); // 入力値の取得 const date = document.getElementById('date').value; const time = document.getElementById('time').value; const lat = document.getElementById('lat').value; const lng = document.getElementById('lng').value; try { resDiv.innerHTML = "計算中..."; // RailwayのURLに向けてデータを送る const resp = await fetch(`/api/horoscope?date=${date}&time=${time}&lat=${lat}&lng=${lng}`); const data = await resp.json(); if (data.error) { resDiv.innerHTML = "エラーが発生しました。"; return; } // 結果の表示 resDiv.innerHTML = ` ☀️ 太陽: ${data.sunPos}
🌙 月: ${data.moonPos}
♂️ 火星: ${data.marsPos}
⬆️ ASC: ${data.ascPos} `; } catch (error) { // ここが抜けていると SyntaxError になります console.error(error); resDiv.innerHTML = "通信エラーが発生しました。"; } }
